import java.util.Scanner;

public class main {
    public static void main(String[] args) {
        Scanner leer = new Scanner(System.in);
        double[] notas = new double[20];
        String[] nombre = new String[20];
        int n = 0, i = 0, opcion = 0;

        while (opcion != 7) {
            System.out.println("..... MENÚ PRINCIPAL .....");
            System.out.println("1. Registrar estudiantes");
            System.out.println("2. Mostrar estudiantes y notas");
            System.out.println("3. Mostrar promedio del curso");
            System.out.println("4. Mostrar mejor y peor estudiante");
            System.out.println("5. Mostrar cantidad de aprobados y reprobados");
            System.out.println("6. Ordenar estudiantes por nota");
            System.out.println("7. Mostrar programadores");
            System.out.println("8. Salir");
            System.out.println("");
            System.out.print("Escoger Opción: ");

            opcion = leer.nextInt(); // Se solicita la opción en cada vuelta

            switch (opcion) {

                case 1:
                    // PASO 2: Preguntamos el valor de 'n' antes de llenar para que el for funcione
                    System.out.print("¿Cuántos estudiantes deseas ingresar?: ");
                    n = leer.nextInt();
                    llenarvector(nombre, notas, n, leer);
                    break;

                case 2:
                    mostrarvector(nombre, notas, n);
                    break;

                case 3:
                    double resultado = promediovector(notas, n);
                    System.out.println("----- PROMEDIO CURSO -----");
                    System.out.println("El promedio es: " + resultado + "\n");
                    break;

                case 4:
                    System.out.println("----- MEJOR ESTUDIANTE -----");
                    System.out.println("Nombre: " + mejorvector(nombre, notas, n) + "\n");
                    System.out.println("----- PEOR ESTUDIANTE -----");
                    System.out.println("Nombre: " + peorvector(nombre, notas, n) + "\n");
                    break;

                case 5:
                    bien_malvector(notas, n);
                break;

                case 6:
                    ordenarvector(nombre, notas, n);
                    break;

                case 7:
                    mostrar_programadores();
                    break;

                case 8:
                    System.out.println("Programa finalizado");
                    System.out.println("--¡ Vuelva Pronto !--");
                    break;
                default:
                    System.out.println("!-- La opción es incorrecta --¡");
                    break;
            }

        }
    }

    // Método llenarvector
    public static void llenarvector(String[] nombre, double[] notas, int n, Scanner leer) {
        int i = 0;
        System.out.println("..... REGISTRAR ESTUDIANTE .....");
        for (i = 0; i < n; i++) {
            System.out.println("----------------------------------");
            System.out.println("Estudiante " + (i + 1));
            leer.nextLine();
            System.out.printf("Ingrese nombre del estudiante: ");
            nombre[i] = leer.nextLine();
            System.out.printf("Ingrese nota (Ccon la coma '',''): ");
            notas[i] = leer.nextDouble();
        }
        System.out.println("");
    } // Fin Llenar Vector

    // Mostrar Vector
    public static void mostrarvector(String[] nombre, double[] notas, int n) {
        int i = 0;
        System.out.println("..... MOSTRAR ESTUDIANTE .....");
        for (i = 0; i < n; i++) {
            System.out.println("----------------------------------");
            System.out.println("Estudiante " + (i + 1));
            System.out.print("Nombre: " + nombre[i] + " \n"); // Cambiado printf por print para evitar errores de formato básico
            System.out.println("Nota: " + notas[i] + " "); // Cambiado printf por print para evitar errores de formato básico
        }
        System.out.println("");
    } // Fin Mostrar Vector

    // Función Promedio vector
    public static double promediovector(double[] notas, double n) {
        double suma = 0, promedio = 0;
        int i = 0;
        for (i = 0; i < n; i++) {
            suma = suma + notas[i];
            promedio = suma / n;
        }
        return promedio;
    } // Fin Función - Promedio Vector

    // Función Mejor vector
    public static String mejorvector(String[] nombre, double[] notas, int n){
        String nombre_mayor="";
        double mayor = 0;

        for (int i = 0; i < n; i++) {
            if (notas[i] > mayor){
                mayor = notas[i];
                nombre_mayor = nombre[i];
            } 
        }
        return nombre_mayor;
    }// Fin Función - Mejor Vector

    // Función Peor vector
    public static String peorvector(String[] nombre, double[] notas, int n){
        String nombre_menor="";
        double menor = 50;
        for (int i = 0; i < n; i++) {
            if (notas[i] < menor){
                menor = notas[i];
                nombre_menor = nombre[i];
            } 
        }
        return nombre_menor;
    }// Fin Función - Peor Vector

    // Función Bien y Mal vector
    public static void bien_malvector(double[] notas, int n){

        int cant_bien = 0;
        int cant_mal = 0;

        for (int i = 0; i < n; i++) {

            if (notas[i] >= 3.0){
                cant_bien++;

            } else {
                cant_mal++;
            }
        }

        System.out.println("----- RESULTADOS -----");
        System.out.println("Estudiantes Aprobados: " + cant_bien);
        System.out.println("Estudiantes Reprobados: " + cant_mal);
    }// Fin Función - Bien y Mal vector

    // Función Orden vector
    public static void ordenarvector(String[] nombre, double[] notas, int n){
        double nota_vector = 0;
        String nombre_vector = "";

        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - 1 - i; j++) {

                if (notas[j] < notas[j + 1]) {
                    nota_vector= notas[j];
                    notas[j] = notas[j + 1];
                    notas[j + 1] = nota_vector;

                    nombre_vector = nombre[j];
                    nombre[j] = nombre[j + 1];
                    nombre[j + 1] = nombre_vector;
                }
            }
        }
        System.out.println("¡Vector ordenado correctamente!");
    System.out.println("Usa la opción 2 del menú para ver el resultado.");
    }// Fin Función - Orden vector
